function printBinaryPyramid($rows)
{
    for ($i = 1; $i <= $rows; $i++) {
        if ($i % 2 == 0) {
            $num = 0;
        } else {
            $num = 1;
        }
        for ($j = 1; $j <= $i; $j++) {
            echo $num . " ";
            if ($num == 1) {

                $num = 1;
