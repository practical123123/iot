def next_date(day, month, year):
    if not (1 <= month <= 12):
        return "value of month not in the range 1…12"
    if not (1 <= day <= 31):
        return "value of day not in the range 1…31"
    if not (1900 <= year <= 2025):
        return "value of year not in the range 1900…2025"


    month_days = [31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31]


    if (year % 4 == 0 and year % 100 != 0) or (year % 400 == 0):
        month_days[1] = 29


    if day > month_days[month - 1]:
        return "invalid day for the given month"


    day += 1
    if day > month_days[month - 1]:
        day = 1
        month += 1


    if month > 12:
        month = 1
        year += 1


    return f"{day:02d}/{month:02d}/{year}"

# Photo Upload Validation Program
def validate_photo(file_name, file_size_kb, resolution):
    """ Validates the uploaded photo based on:
    - Format: must be .jpg
    - Size: must be < 32 KB
    - Resolution: must be 137x177
    """
    format_ok = file_name.lower().endswith(".jpg")
    size_ok = file_size_kb < 32
    resolution_ok = resolution == (137, 177)
    
    if format_ok and size_ok and resolution_ok:
        return "OK - Photo Uploaded"
    elif format_ok and size_ok and not resolution_ok:
        return "E1 - Error: resolution mismatch"
    elif format_ok and not size_ok and resolution_ok:
        return "E2 - Error: size mismatch"
    elif format_ok and not size_ok and not resolution_ok:
        return "E3 - Error: size and resolution mismatch"
    elif not format_ok and size_ok and resolution_ok:
        return "E4 - Error: format mismatch"
    elif not format_ok and size_ok and not resolution_ok:
        return "E5 - Error: format and resolution mismatch"
    elif not format_ok and not size_ok and resolution_ok:
        return "E6 - Error: format and size mismatch"
    else:  # not format_ok and not size_ok and not resolution_ok
        return "E7 - Error: format, size, and resolution mismatch"

def calculate_discount(units_ordered, cod, wholesale):
    """
    Calculate discount rate for merchandise orders.
    
    Parameters:
    - units_ordered: int
    - cod: bool, True if Cash on Delivery
    - wholesale: bool, True if Wholesale Outlet
    
    Returns:
    - discount_rate: int (0, 2, 4, 6)
    """
    discount = 0
    # Wholesale discount
    if wholesale:
        discount += 2
    # COD discount
    if cod:
        discount += 2
    # Bulk order discount
    if units_ordered >= 50:

    return discount

def money_transfer(account_approved, otp_matched, sufficient_money):
    # If account is not approved, block immediately (safety rule)
    if not account_approved:
        return "Block Transaction"


    # If OTP fails, it's suspicious -> block
    if not otp_matched:
        return "Block Transaction"


    # If money is insufficient -> show message
    if not sufficient_money:
        return "Insufficient Amount"


    # All good -> transfer money
    return "Transfer Money"

Cyclomatic Complexity
A) Path Testing (Compute Cyclomatic Complexity)
Cyclomatic complexity is a metric that measures the complexity of a program by counting its decision points. It determines the number of unique paths through the code, indicating how complex the logic is. Lower complexity suggests simpler and more manageable code, reducing the chances of errors and making the program easier to maintain and modify. In essence, it helps assess the code’s readability and the risk associated with making changes.
If a piece of source code contains no control-flow statements, then its cyclomatic complexity is 1, meaning the code has a single execution path.
Similarly, if the source code contains one if condition, then the cyclomatic complexity will be 2, because there are two possible paths: one for the condition being true and another for it being false.
Formula for calculating Cyclomatic Complexity
M = E-N+2P
Where 
E = the number of edges in the control flow graph
N = the number of nodes in the control flow graph
P = the number of connected components (usually 1)
